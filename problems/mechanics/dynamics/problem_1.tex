%<*problemtag>
\probinit
\probname{}

\probsource{Course of Theoretical mechanics at Charles university 2020}
\probtopics{Dynamics}

\probtask{Point mass $m$ is repeled by force $F$, proportional to the third power of distance from the origin of space coordinate $x$. Solve the motion of the point mass, if it is released from rest from the point $x_0$.
}
%</problemtag>
%<*solutiontag>
\probsolution{

}
%</solutiontag>
%<*answertag>
\probanswer{
    \[
        x(t)=\sqrt{x_0^2+\frac{kt^2}{mx_0^2}}
    \]
}
%</answertag>
