\probname{}

\probsource{Course of Theoretical mechanics at Charles university 2020}
\probtopics{Dynamics}

\probtask{Boat of mass $m$ is moving with the speed $v_0$ and stop the engine. Find time $t_s$, after which the boat stops, and the distance $x_s$ it traveled, if it is decelerating by the force $F=-b\exp(av)$ where $a$ and $b$ are possitive constants.
}
%</problemtag>
%<*solutiontag>
\probsolution{
}
%</solutiontag>
%<*answertag>
\probanswer{
    \[
    \begin{split}
        t_s&=\frac{m}{ab}\left(1-e^{-av_0}\right)\\
        x_s&=\frac{m}{a^2b}=\frac{m}{a^2b}\left[1-\left(1+av_0\right)e^{-av_0}\right]
    \end{split}
    \]
}
%</answertag>
