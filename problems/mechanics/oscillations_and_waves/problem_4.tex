\probtask{Solve the motion of small object falling through the diameter of earth. Consider denisity, radius $R$ and mass $M$ of earth to be constant.
}
%</problemtag>
%<*solutiontag>
\probsolution{

}
%</solutiontag>
%<*answertag>
\probanswer{
    \[
        x(t)=R\cos(\omega t)\quad\text{where}\;\omega=\sqrt{GM/R^3}
    \]
}
%</answertag>
